```typescript
"use client";

import { createContext, useContext } from "react";

interface OnboardingContextValue {
  stepIndex: number;
  totalSteps: number;
  goNext: () => void;
  goBack: () => void;
  isTourMode: boolean;
  canGoBack: boolean;
  formData?: ProfileFormData;
  setFormData?: (data: ProfileFormData) => void;
}

const OnboardingContext = createContext<OnboardingContextValue | null>(null);

export function useOnboarding() {
  const ctx = useContext(OnboardingContext);
  if (!ctx) throw new Error("useOnboarding must be used within OnboardingProvider");
  return ctx;
}

export { OnboardingContext };
```

```typescript
"use client";

import { useState } from "react";
import { OnboardingContext } from "./onboarding-context";
import { WelcomeStep } from "./steps/welcome-step";
import { MeetTeamStep } from "./steps/meet-team-step";
import { ProfileStep } from "./steps/profile-step";
import { ReadyStep } from "./steps/ready-step";
import { AnimatePresence } from "framer-motion";

export function OnboardingModal() {
  const [stepIndex, setStepIndex] = useState(0);
  const [isTourMode, setIsTourMode] = useState(false);

  const activeSteps = isTourMode ? TOUR_STEPS : ALL_STEPS;
  const currentStep = activeSteps[stepIndex];

  const contextValue = {
    stepIndex,
    totalSteps: activeSteps.length,
    goNext: () => setStepIndex(prev => Math.min(prev + 1, activeSteps.length - 1)),
    goBack: () => setStepIndex(prev => Math.max(prev - 1, 0)),
    isTourMode,
    canGoBack: stepIndex > 0,
  };

  return (
    <OnboardingContext.Provider value={contextValue}>
      <div className="...">
        <AnimatePresence mode="wait">
          {currentStep.id === "welcome" && <WelcomeStep key="welcome" />}
          {currentStep.id === "meet-team" && <MeetTeamStep key="meet-team" />}
          {currentStep.id === "profile" && <ProfileStep key="profile" />}
          {currentStep.id === "ready" && <ReadyStep key="ready" />}
        </AnimatePresence>
      </div>
    </OnboardingContext.Provider>
  );
}
```
